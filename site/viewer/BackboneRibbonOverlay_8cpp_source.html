25// Amino acid colors — muted, grouped by physicochemical property.
26// Hydrophobic: warm earth tones. Polar: soft blue-green. Charged: muted
27// warm (positive) or cool (negative). Aromatic: dusty purple/mauve.
28// Special: distinct but not loud.
29static void ResidueColor(AminoAcid aa, unsigned char rgb[3]) {
30 switch (aa) {
31 // Hydrophobic — warm tans, khaki, olive
32 case AminoAcid::ALA: rgb[0]=195; rgb[1]=180; rgb[2]=155; break; // warm sand
33 case AminoAcid::VAL: rgb[0]=175; rgb[1]=165; rgb[2]=130; break; // khaki
34 case AminoAcid::LEU: rgb[0]=185; rgb[1]=170; rgb[2]=135; break; // light khaki
35 case AminoAcid::ILE: rgb[0]=165; rgb[1]=155; rgb[2]=120; break; // olive tan
36 case AminoAcid::MET: rgb[0]=190; rgb[1]=180; rgb[2]=110; break; // muted gold
37
38 // Aromatic — dusty purple/mauve (distinct from everything else)
39 case AminoAcid::PHE: rgb[0]=155; rgb[1]=120; rgb[2]=155; break; // dusty mauve
40 case AminoAcid::TRP: rgb[0]=140; rgb[1]=105; rgb[2]=150; break; // deeper mauve
41 case AminoAcid::TYR: rgb[0]=160; rgb[1]=130; rgb[2]=160; break; // light mauve
42 case AminoAcid::HIS: rgb[0]=150; rgb[1]=140; rgb[2]=180; break; // slate lavender
43
44 // Polar — soft blue-green
45 case AminoAcid::SER: rgb[0]=140; rgb[1]=185; rgb[2]=175; break; // sage
46 case AminoAcid::THR: rgb[0]=150; rgb[1]=190; rgb[2]=170; break; // light sage
47 case AminoAcid::ASN: rgb[0]=130; rgb[1]=175; rgb[2]=185; break; // soft teal
48 case AminoAcid::GLN: rgb[0]=140; rgb[1]=180; rgb[2]=190; break; // light teal

73 // Build the polydata arrays that vtkProteinRibbonFilter expects.
74 // Required arrays (from VTK source):
75 // "atom_types" vtkStringArray PDB atom name ("CA","O","N","C",...)
76 // "atom_type" vtkIdTypeArray atomic number
77 // "residue" vtkIdTypeArray residue sequence number
78 // "chain" vtkUnsignedCharArray chain ID as char
79 // "secondary_structures" vtkUnsignedCharArray 'h','s','c' per atom
80 // "secondary_structures_begin" vtkUnsignedCharArray 1 at SS start
81 // "secondary_structures_end" vtkUnsignedCharArray 1 at SS end
82 // "ishetatm" vtkUnsignedCharArray 0 for protein atoms
83
84 vtkIdType nAtoms = static_cast<vtkIdType>(protein.AtomCount());
85
86 vtkNew<vtkPoints> points;
87 points->SetNumberOfPoints(nAtoms);
88
89 vtkNew<vtkStringArray> atomTypes;
90 atomTypes->SetName("atom_types");
91 atomTypes->SetNumberOfValues(nAtoms);
92
93 vtkNew<vtkIdTypeArray> atomType;
94 atomType->SetName("atom_type");
95 atomType->SetNumberOfValues(nAtoms);
96
97 vtkNew<vtkIdTypeArray> residue;
98 residue->SetName("residue");
99 residue->SetNumberOfValues(nAtoms);
100
101 vtkNew<vtkUnsignedCharArray> chain;
102 chain->SetName("chain");
103 chain->SetNumberOfValues(nAtoms);
104
105 vtkNew<vtkUnsignedCharArray> ss;
106 ss->SetName("secondary_structures");
107 ss->SetNumberOfValues(nAtoms);
108
109 vtkNew<vtkUnsignedCharArray> ssBegin;
110 ssBegin->SetName("secondary_structures_begin");
111 ssBegin->SetNumberOfValues(nAtoms);
112
113 vtkNew<vtkUnsignedCharArray> ssEnd;
114 ssEnd->SetName("secondary_structures_end");
115 ssEnd->SetNumberOfValues(nAtoms);
116
117 vtkNew<vtkUnsignedCharArray> ishetatm;
118 ishetatm->SetName("ishetatm");
119 ishetatm->SetNumberOfValues(nAtoms);
120
121 bool hasDssp = conf.HasResult<DsspResult>();
122 const DsspResult* dssp = hasDssp ? &conf.Result<DsspResult>() : nullptr;
123
124 for (vtkIdType i = 0; i < nAtoms; ++i) {
125 size_t si = static_cast<size_t>(i);
126 const auto& id = protein.AtomAt(si);
127 const auto& res = protein.ResidueAt(id.residue_index);
128 Vec3 pos = conf.AtomAt(si).Position();
129
130 points->SetPoint(i, pos.x(), pos.y(), pos.z());
131 atomTypes->SetValue(i, id.pdb_atom_name);
132 atomType->SetValue(i, static_cast<vtkIdType>(AtomicNumberForElement(id.element)));
133 residue->SetValue(i, static_cast<vtkIdType>(res.sequence_number));
134 chain->SetValue(i, res.chain_id.empty() ? 'A' : res.chain_id[0]);
135 ishetatm->SetValue(i, 0);
136
137 // Secondary structure from DSSP: H=helix, E=sheet, else coil
138 unsigned char ssChar = 'c';
139 if (dssp) {
140 char dsspCode = dssp->SecondaryStructure(id.residue_index);
141 if (dsspCode == 'H' || dsspCode == 'G' || dsspCode == 'I')
142 ssChar = 'h';
143 else if (dsspCode == 'E' || dsspCode == 'B')
144 ssChar = 's';
145 }
146 ss->SetValue(i, ssChar);
147 ssBegin->SetValue(i, 0);
148 ssEnd->SetValue(i, 0);
149 }
150
151 // Mark SS begin/end transitions
152 if (dssp) {
153 unsigned char prevSS = 'c';
154 for (vtkIdType i = 0; i < nAtoms; ++i) {
155 unsigned char curSS = ss->GetValue(i);
156 if (curSS != prevSS) {
157 if (curSS != 'c') ssBegin->SetValue(i, 1);
158 if (prevSS != 'c' && i > 0) ssEnd->SetValue(i - 1, 1);
159 }
160 prevSS = curSS;
161 }
162 // Last atom ends its SS
163 if (nAtoms > 0 && ss->GetValue(nAtoms - 1) != 'c')
164 ssEnd->SetValue(nAtoms - 1, 1);
165 }
166
167 vtkNew<vtkPolyData> pd;
168 pd->SetPoints(points);
169 pd->GetPointData()->AddArray(atomTypes);
170 pd->GetPointData()->AddArray(atomType);
171 pd->GetPointData()->AddArray(residue);
172 pd->GetPointData()->AddArray(chain);
173 pd->GetPointData()->AddArray(ss);
174 pd->GetPointData()->AddArray(ssBegin);
175 pd->GetPointData()->AddArray(ssEnd);
176 pd->GetPointData()->AddArray(ishetatm);
177
178 // Run the ribbon filter
179 vtkNew<vtkTrivialProducer> producer;
180 producer->SetOutput(pd);
181
182 vtkNew<vtkProteinRibbonFilter> ribbon;
183 ribbon->SetInputConnection(producer->GetOutputPort());
184 ribbon->SetCoilWidth(0.5f);
185 ribbon->SetHelixWidth(1.8f);
186 ribbon->SetSubdivideFactor(20);
187 ribbon->SetDrawSmallMoleculesAsSpheres(false);
188 ribbon->Update();
189
190 vtkPolyData* ribbonOutput = ribbon->GetOutput();
191 if (!ribbonOutput || ribbonOutput->GetNumberOfPoints() == 0)
192 return;
193
194 // Replace the filter's default colors with per-residue colors.
195 // The filter walks CAs in input order, breaks at chain/residue gaps,
196 // and builds strips. Within each strip, each CA contributes subdiv
197 // subdivision points, each producing 2 border points (left/right of
198 // ribbon) with the same color. We replicate the CA walk to build
199 // per-segment residue color lists, then step through the output RGB
200 // array in the same order.
201 vtkUnsignedCharArray* ribbonRGB = vtkArrayDownCast<vtkUnsignedCharArray>(
202 ribbonOutput->GetPointData()->GetScalars("RGB"));
203 if (ribbonRGB) {
204 int subdiv = ribbon->GetSubdivideFactor();
205
206 // Build per-segment CA lists using the same chain-break logic as
207 // vtkProteinRibbonFilter::RequestData (lines 168-189 of VTK source)
208 struct Segment { std::vector<AminoAcid> residues; };
209 std::vector<Segment> segments;
210 segments.push_back({});
211
212 unsigned char prevChainVal = 0;
213 vtkIdType prevResiVal = -999;
214 bool firstCA = true;
215
216 for (vtkIdType i = 0; i < nAtoms; ++i) {
217 std::string atype = atomTypes->GetValue(i);
218 if (ishetatm->GetValue(i)) continue;
219 if (atype == "CA") {
220 unsigned char ch = chain->GetValue(i);
221 vtkIdType ri = residue->GetValue(i);
222 if (!firstCA && (ch != prevChainVal || ri != prevResiVal + 1)) {
223 // Chain break — start new segment
224 if (segments.back().residues.size() >= 2)
225 segments.push_back({});
226 else
227 segments.back().residues.clear();
228 }
229 size_t si = static_cast<size_t>(i);
230 segments.back().residues.push_back(
231 protein.ResidueAt(protein.AtomAt(si).residue_index).type);
232 prevChainVal = ch;
233 prevResiVal = ri;
234 firstCA = false;
235 }
236 }
237
238 // Walk the output RGB array, assigning residue colors per segment.
239 // Each segment with nCA >= 2 produces a strip with
240 // 2 * ((nCA-1)*subdiv + 1) RGB tuples.
241 vtkIdType rgbIdx = 0;
242 vtkIdType totalTuples = ribbonRGB->GetNumberOfTuples();
243
244 for (const auto& seg : segments) {
245 int nCA = static_cast<int>(seg.residues.size());
246 if (nCA < 2) continue;
247 int len = (nCA - 1) * subdiv + 1; // subdivision points per strip
248
249 for (int si = 0; si < len; ++si) {
250 int caIdx = static_cast<int>(floor(0.5 + si / static_cast<double>(subdiv)));
251 if (caIdx >= nCA) caIdx = nCA - 1;
252
253 unsigned char rgb[3];
254 ResidueColor(seg.residues[caIdx], rgb);
255
256 // Two border points per subdivision point
257 for (int k = 0; k < 2; ++k) {
258 if (rgbIdx < totalTuples)
259 ribbonRGB->SetTypedTuple(rgbIdx++, rgb);
260 }
261 }
262 }
263 }
264
265 vtkNew<vtkPolyDataMapper> mapper;
266 mapper->SetInputData(ribbonOutput);
267 mapper->SetScalarModeToUsePointData();
268 mapper->SelectColorArray("RGB");
269 mapper->SetColorModeToDirectScalars();
270
271 actor_ = vtkSmartPointer<vtkActor>::New();
272 actor_->SetMapper(mapper);
273 actor_->GetProperty()->SetInterpolationToPhong();
274 actor_->GetProperty()->SetOpacity(0.95);
275
276 renderer_->AddActor(actor_);
277}
