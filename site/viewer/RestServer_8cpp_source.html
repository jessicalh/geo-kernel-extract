37RestServer::RestServer(MainWindow* mainWindow, quint16 port, QObject* parent)
38 : QObject(parent)
39 , server_(new QTcpServer(this))
40 , mainWindow_(mainWindow)
41{
42 connect(server_, &QTcpServer::newConnection,
43 this, &RestServer::onNewConnection);
44
45 // Try requested port, then up to 9 more (9147, 9148, ..., 9156)
46 for (quint16 p = port; p < port + 10; ++p) {
47 if (server_->listen(QHostAddress::LocalHost, p)) {
48 actualPort_ = p;
49 std::cout << "REST server listening on localhost:" << p;
50 if (p != port) std::cout << " (requested " << port << " was in use)";
51 std::cout << "\n";
52 return;
53 }
54 }
55 std::cerr << "REST server: could not bind ports " << port
56 << "-" << (port + 9) << "\n";
57}

59void RestServer::onNewConnection() {
60 while (server_->hasPendingConnections()) {
61 auto* socket = server_->nextPendingConnection();
62 clients_.append(socket);
63 connect(socket, &QTcpSocket::readyRead,
64 this, &RestServer::onReadyRead);
65 connect(socket, &QTcpSocket::disconnected,
66 this, &RestServer::onDisconnected);
67 }

70void RestServer::onReadyRead() {
71 auto* socket = qobject_cast<QTcpSocket*>(sender());
72 if (!socket) return;
73
74 while (socket->canReadLine()) {
75 QByteArray line = socket->readLine().trimmed();
76 if (line.isEmpty()) continue;
77
78 QJsonParseError err;
79 QJsonDocument doc = QJsonDocument::fromJson(line, &err);
80 if (err.error != QJsonParseError::NoError || !doc.isObject()) {
81 QJsonObject resp;
82 resp["ok"] = false;
83 resp["error"] = QString("JSON parse error: %1").arg(err.errorString());
84 sendResponse(socket, resp);
85 continue;
86 }
87
88 QJsonObject result = dispatch(doc.object());
89 sendResponse(socket, result);
90 }
91}

143QJsonObject RestServer::cmdStatus() {
144 QJsonObject resp;
145 resp["ok"] = true;
146 QJsonObject result;
147
148 // Report computation state
149 result["computing"] = (mainWindow_->workerThread_ != nullptr &&
150 mainWindow_->workerThread_->isRunning());
151
152 // Read directly from the library protein model
153 auto& protein = mainWindow_->protein_;
154 if (protein) {
155 const auto& conf = protein->Conformation();
156 result["protein"] = QString::fromStdString(mainWindow_->currentProteinId_);
157 result["n_atoms"] = (int)protein->AtomCount();
158 result["n_rings"] = (int)protein->RingCount();
159 result["n_residues"] = (int)protein->ResidueCount();
160
161 // Tier counts: read from library (set by PredictionResult)
162 int nReport = 0, nPass = 0, nSilent = 0;
163 for (size_t i = 0; i < conf.AtomCount(); ++i) {
164 switch (conf.AtomAt(i).tier) {
165 case HeuristicTier::REPORT: nReport++; break;
166 case HeuristicTier::PASS: nPass++; break;
167 case HeuristicTier::SILENT: nSilent++; break;
168 }
169 }
170 result["n_report"] = nReport;
171 result["n_pass"] = nPass;
172 result["n_silent"] = nSilent;
173 } else {
174 result["protein"] = "";
175 result["n_atoms"] = 0;
176 result["n_rings"] = 0;
177 result["n_residues"] = 0;
178 result["n_report"] = 0;
179 result["n_pass"] = 0;
180 result["n_silent"] = 0;
181 }
182
183 result["overlay_mode"] = 0; // overlay modes removed
184 result["n_field_grids"] = (int)mainWindow_->fieldGrids_.size();
185 result["n_butterfly_grids"] = (int)mainWindow_->butterflyFields_.size();
186 result["field_grid_overlay"] = (mainWindow_->fieldGridOverlay_ != nullptr);
187 result["butterfly_overlay"] = (mainWindow_->butterflyOverlay_ != nullptr);
188 if (!mainWindow_->fieldGrids_.empty()) {
189 double gmin = 1e30, gmax = -1e30;
190 int nz = 0;
191 for (const auto& g : mainWindow_->fieldGrids_) {
192 for (double v : g.T0) {
193 if (v < gmin) gmin = v;
194 if (v > gmax) gmax = v;
195 if (std::abs(v) > 1e-10) nz++;
196 }
197 }
198 result["grid_T0_min"] = gmin;
199 result["grid_T0_max"] = gmax;
200 result["grid_T0_nonzero"] = nz;
201 }
202 resp["result"] = result;
203 return resp;
204}

416QJsonObject RestServer::cmdLookAtRing(const QJsonObject& cmd) {
417 auto& protein = mainWindow_->protein_;
418 if (!protein)
419 return QJsonObject{{"ok", false}, {"error", "no protein loaded"}};
420
421 int ringIdx = cmd["ring"].toInt(-1);
422 if (ringIdx < 0 || ringIdx >= (int)protein->RingCount())
423 return QJsonObject{{"ok", false}, {"error", "invalid ring index"}};
424
425 const auto& conf = protein->Conformation();
426 const auto& geo = conf.ring_geometries[ringIdx];
427 const auto& ring = protein->RingAt(ringIdx);
428
429 Vec3 center = geo.center;
430 Vec3 normal = geo.normal.normalized();
431
432 // Build orthonormal basis in ring plane
433 Vec3 arbitrary = (std::abs(normal.x()) < 0.9) ? Vec3(1,0,0) : Vec3(0,1,0);
434 Vec3 u = normal.cross(arbitrary).normalized();
435 Vec3 v = normal.cross(u);
436
437 // View direction: default "side" (perpendicular to normal, sees butterfly lobes)
438 // "top" looks down the normal, "edge" looks along the other in-plane axis
439 QString view = cmd.value("view").toString("side");
440 double distance = cmd.value("distance").toDouble(15.0);
441
442 Vec3 camDir;
443 Vec3 upVec;
444 if (view == "top") {
445 camDir = normal;
446 upVec = u;
447 } else if (view == "edge") {
448 camDir = v;
449 upVec = normal;
450 } else { // "side"
451 camDir = u;
452 upVec = normal;
453 }
454
455 Vec3 camPos = center + distance * camDir;
456
457 auto* camera = mainWindow_->renderer_->GetActiveCamera();
458 camera->SetPosition(camPos.x(), camPos.y(), camPos.z());
459 camera->SetFocalPoint(center.x(), center.y(), center.z());
460 camera->SetViewUp(upVec.x(), upVec.y(), upVec.z());
461 camera->OrthogonalizeViewUp();
462 mainWindow_->renderer_->ResetCameraClippingRange();
463 mainWindow_->renderWindow_->Render();
464
465 QJsonObject result;
466 result["ring_index"] = ringIdx;
467 result["ring_type"] = QString::fromStdString(ring.TypeName());
468 result["center"] = QJsonArray{center.x(), center.y(), center.z()};
469 result["normal"] = QJsonArray{normal.x(), normal.y(), normal.z()};
470 result["view"] = view;
471 return QJsonObject{{"ok", true}, {"result", result}};
472}

474QJsonObject RestServer::cmdLookAtAtom(const QJsonObject& cmd) {
475 auto& protein = mainWindow_->protein_;
476 if (!protein)
477 return QJsonObject{{"ok", false}, {"error", "no protein loaded"}};
478
479 int atomIdx = cmd["atom"].toInt(-1);
480 if (atomIdx < 0 || atomIdx >= (int)protein->AtomCount())
481 return QJsonObject{{"ok", false}, {"error", "invalid atom index"}};
482
483 const auto& conf = protein->Conformation();
484 Vec3 pos = conf.AtomAt(atomIdx).Position();
485 double distance = cmd.value("distance").toDouble(15.0);
486
487 // Keep current camera direction, just re-center on this atom
488 auto* camera = mainWindow_->renderer_->GetActiveCamera();
489 double camPos[3], foc[3];
490 camera->GetPosition(camPos);
491 camera->GetFocalPoint(foc);
492 Vec3 dir(camPos[0] - foc[0], camPos[1] - foc[1], camPos[2] - foc[2]);
493 dir.normalize();
494
495 Vec3 newCam = pos + distance * dir;
496 camera->SetFocalPoint(pos.x(), pos.y(), pos.z());
497 camera->SetPosition(newCam.x(), newCam.y(), newCam.z());
498 camera->OrthogonalizeViewUp();
499 mainWindow_->renderer_->ResetCameraClippingRange();
500 mainWindow_->renderWindow_->Render();
501
502 const auto& id = protein->AtomAt(atomIdx);
503 const auto& res = protein->ResidueAt(id.residue_index);
504 QJsonObject result;
505 result["atom_index"] = atomIdx;
506 result["element"] = QString::fromStdString(SymbolForElement(id.element));
507 result["pdb_name"] = QString::fromStdString(id.pdb_atom_name);
508 result["residue"] = QString("%1-%2").arg(
509 QString::fromStdString(ThreeLetterCodeForAminoAcid(res.type)))
510 .arg(res.sequence_number);
511 return QJsonObject{{"ok", true}, {"result", result}};

540QJsonObject RestServer::cmdGetLog(const QJsonObject& cmd) {
541 QString text = mainWindow_->logText_->toPlainText();
542 QStringList all = text.split('\n');
543 int total = all.size();
544
545 // "lines":N — last N lines (convenience for tail)
546 // "first":F, "last":L — specific range [F, L] inclusive, 0-based
547 // No args — everything
548 int first = 0, last = total - 1;
549 if (cmd.contains("lines")) {
550 int n = cmd["lines"].toInt(50);
551 first = std::max(0, total - n);
552 } else if (cmd.contains("first") || cmd.contains("last")) {
553 first = cmd.value("first").toInt(0);
554 last = cmd.value("last").toInt(total - 1);
555 }
556 first = std::max(0, std::min(first, total - 1));
557 last = std::max(first, std::min(last, total - 1));
558
559 QJsonArray lines;
560 for (int i = first; i <= last; ++i)
561 lines.append(all[i]);
562
563 QJsonObject result;
564 result["total_lines"] = total;
565 result["first"] = first;
566 result["last"] = last;
567 result["returned"] = lines.size();
568 result["lines"] = lines;
569 return QJsonObject{{"ok", true}, {"result", result}};
570}
