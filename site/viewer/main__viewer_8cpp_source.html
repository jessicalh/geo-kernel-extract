44// ============================================================================
45// Extract --rest-port from argv before JobSpec parsing.
46// JobSpec ignores unknown flags, but we need to pull this out for Qt.
47// ============================================================================
48
49static quint16 ExtractRestPort(int argc, char* argv[]) {
50 for (int i = 1; i < argc - 1; ++i) {
51 if (std::strcmp(argv[i], "--rest-port") == 0)
52 return static_cast<quint16>(std::atoi(argv[i + 1]));
53 }
54 return 9147; // default
55}

65 // 2. Library environment + logging

71 // 3. Parse command line via JobSpec
72 // --output is optional for the viewer (no output = just visualise).
73 // --rest-port is viewer-specific, extracted separately.
74 auto spec = nmr::ParseJobSpec(argc, argv);
75 quint16 restPort = ExtractRestPort(argc, argv);
76
77 // Handle --help: print both JobSpec usage and viewer extras
78 if (spec.mode == nmr::JobMode::None && spec.error.empty()) {
79 nmr::PrintJobSpecUsage(argv[0]);
80 fprintf(stderr,
81 "\nViewer options:\n"
82 " --rest-port PORT REST API port (default 9147, 0 to disable)\n"
83 "\n"
84 " The viewer accepts all the same modes as nmr_extract.\n"
85 " --output is optional: omit it to just visualise.\n");
86 return 0;
87 }
88
89 // Parse error
90 if (!spec.Ok()) {
91 fprintf(stderr, "ERROR: %s\n\n", spec.error.c_str());
92 nmr::PrintJobSpecUsage(argv[0]);
93 return 1;
94 }
95
96 // Validate file existence
97 if (!nmr::ValidateJobSpec(spec)) {
98 QMessageBox::critical(nullptr, "Validation Error",
99 QString::fromStdString(spec.error));
100 return 1;
101 }
102
103 // Print warnings
104 for (const auto& w : spec.warnings)
105 fprintf(stderr, "WARNING: %s\n", w.c_str());
106
107 // Load calculator config if specified
108 if (!spec.config_path.empty())
109 nmr::CalculatorConfig::Load(spec.config_path);
110
111 // 4. Register metatypes for cross-thread signals
112 qRegisterMetaType<nmr::JobSpec>("nmr::JobSpec");
113 qRegisterMetaType<ComputeResult>("ComputeResult");
114
115 // 5. Create and show window
116 MainWindow window;
117 QObject::connect(&app, &QCoreApplication::aboutToQuit, &window, &MainWindow::shutdown);
118 window.show();
119
120 // 5b. REST API server
121 RestServer* restServer = nullptr;
122 if (restPort > 0) {
123 restServer = new RestServer(&window, restPort, &app);
124 if (restServer->actualPort() > 0) {
125 window.setWindowTitle(QString("Protein Tensor Viewer — REST :%1")

127 }
128 }
129
130 // 6. Deferred auto-load from JobSpec — all modes supported
131 if (spec.mode != nmr::JobMode::None) {
132 QTimer::singleShot(0, &window, [&window, spec]() {
133 nmr::OperationLog::Info(nmr::LogViewer, "main",
134 "auto-loading from JobSpec");
135 window.loadFromJobSpec(spec);
136 });
137 }
138
139 nmr::OperationLog::Info(nmr::LogViewer, "main", "entering event loop");
