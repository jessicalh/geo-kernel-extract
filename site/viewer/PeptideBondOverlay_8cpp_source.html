20// Build one actor from many line segments, all the same color
21vtkSmartPointer<vtkActor> makeBatchedTubes(
22 const std::vector<std::pair<Vec3,Vec3>>& segments,
23 double r, double g, double b,
24 double radius, double opacity)
25{
26 if (segments.empty()) return nullptr;
27
28 vtkNew<vtkPoints> pts;
29 vtkNew<vtkCellArray> lines;
30
31 for (const auto& [start, end] : segments) {
32 vtkIdType id0 = pts->InsertNextPoint(start.data());
33 vtkIdType id1 = pts->InsertNextPoint(end.data());
34 vtkNew<vtkLine> line;
35 line->GetPointIds()->SetId(0, id0);
36 line->GetPointIds()->SetId(1, id1);
37 lines->InsertNextCell(line);
38 }
39
40 vtkNew<vtkPolyData> pd;
41 pd->SetPoints(pts);
42 pd->SetLines(lines);
43
44 vtkNew<vtkTubeFilter> tube;
45 tube->SetInputData(pd);
46 tube->SetRadius(radius);
47 tube->SetNumberOfSides(8);
48
49 vtkNew<vtkPolyDataMapper> mapper;
50 mapper->SetInputConnection(tube->GetOutputPort());
51
52 auto actor = vtkSmartPointer<vtkActor>::New();
53 actor->SetMapper(mapper);
54 actor->GetProperty()->SetColor(r, g, b);
55 actor->GetProperty()->SetOpacity(opacity);
56 actor->SetForceTranslucent(true);
57 actor->GetProperty()->SetBackfaceCulling(true);
58 return actor;
59}

77void PeptideBondOverlay::buildActors(
78 const Protein& protein,
79 const ProteinConformation& conf)
80{
81 std::vector<std::pair<Vec3,Vec3>> peptideSegments, scSegments, otherSegments;
82
83 for (size_t i = 0; i < protein.BondCount(); ++i) {
84 const Bond& bond = protein.BondAt(i);
85 Vec3 posA = conf.AtomAt(bond.atom_index_a).Position();
86 Vec3 posB = conf.AtomAt(bond.atom_index_b).Position();
87 auto seg = std::make_pair(posA, posB);
88
89 if (bond.IsPeptideBond()) {
90 peptideSegments.push_back(seg);
91 } else if (bond.category == BondCategory::SidechainOther ||
92 bond.category == BondCategory::SidechainCO) {
93 scSegments.push_back(seg);
94 } else {
95 otherSegments.push_back(seg);
96 }
97 }
98
99 // Peptide bonds — red
100 if (auto a = makeBatchedTubes(peptideSegments, 0.9, 0.2, 0.2, tubeRadius_, 0.7)) {
101 renderer_->AddActor(a);
102 bbActors_.push_back(a);
103 }
104 // Other backbone bonds — blue
105 if (auto a = makeBatchedTubes(otherSegments, 0.2, 0.3, 0.9, tubeRadius_, 0.7)) {
106 renderer_->AddActor(a);
107 bbActors_.push_back(a);
108 }
109 // Sidechain bonds — orange
110 if (auto a = makeBatchedTubes(scSegments, 0.9, 0.6, 0.1, tubeRadius_ * 0.8, 0.6)) {
111 renderer_->AddActor(a);
112 scActors_.push_back(a);
113 }
114}
