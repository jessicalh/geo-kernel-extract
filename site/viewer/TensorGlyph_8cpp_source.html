33vtkSmartPointer<vtkActor> TensorGlyph::createGlyph(
34 const nmr::Vec3& center,
35 const nmr::SphericalTensor& st,
36 double scale,
37 double opacity)
38{
39 if (st.T2Magnitude() < 1e-6)
40 return nullptr;
41
42 vtkNew<vtkPoints> points;
43 vtkNew<vtkCellArray> triangles;
44 vtkNew<vtkUnsignedCharArray> colors;
45 colors->SetNumberOfComponents(3);
46 colors->SetName("Colors");
47
48 // Generate sphere vertices
49 int nTheta = THETA_RES;
50 int nPhi = PHI_RES;
51
52 // Store point indices in a grid
53 std::vector<std::vector<int>> grid(nTheta + 1, std::vector<int>(nPhi));
54
55 for (int i = 0; i <= nTheta; ++i) {
56 double theta = M_PI * i / nTheta;
57 for (int j = 0; j < nPhi; ++j) {
58 double phi = 2.0 * M_PI * j / nPhi;
59
60 // f(theta,phi) = sum_m T2_m * Y_2^m(theta,phi)
61 // Real spherical harmonics (Condon-Shortley convention):
62 double ct = std::cos(theta), st2 = std::sin(theta);
63 double cp = std::cos(phi), sp = std::sin(phi);
64 double c2p = std::cos(2*phi), s2p = std::sin(2*phi);
65 // Y_2^{-2} = 0.5*sqrt(15/pi)*sin^2(theta)*sin(2phi)
66 // Y_2^{-1} = 0.5*sqrt(15/pi)*sin(2theta)*sin(phi)
67 // Y_2^{0} = 0.25*sqrt(5/pi)*(3cos^2(theta)-1)
68 // Y_2^{1} = -0.5*sqrt(15/pi)*sin(2theta)*cos(phi)
69 // Y_2^{2} = 0.5*sqrt(15/pi)*sin^2(theta)*cos(2phi)
70 double s15p = 0.5 * std::sqrt(15.0 / M_PI);
71 double s5p = 0.25 * std::sqrt(5.0 / M_PI);
72 double f = st.T2[0] * s15p * st2 * st2 * s2p // m=-2
73 + st.T2[1] * s15p * std::sin(2*theta) * sp // m=-1
74 + st.T2[2] * s5p * (3*ct*ct - 1) // m=0
75 + st.T2[3] * (-s15p) * std::sin(2*theta) * cp // m=1
76 + st.T2[4] * s15p * st2 * st2 * c2p; // m=2
77 double r = std::abs(f) * scale;
78
79 double x = center(0) + r * std::sin(theta) * std::cos(phi);
80 double y = center(1) + r * std::sin(theta) * std::sin(phi);
81 double z = center(2) + r * std::cos(theta);
82
83 int id = (int)points->InsertNextPoint(x, y, z);
84 grid[i][j] = id;
85
86 // Color: red for positive (deshielded), blue for negative (shielded)
87 if (f > 0) {
88 unsigned char c[3] = {200, 50, 50};
89 colors->InsertNextTypedTuple(c);
90 } else {
91 unsigned char c[3] = {50, 50, 200};
92 colors->InsertNextTypedTuple(c);
93 }
94 }
95 }
96
97 // Create triangles
98 for (int i = 0; i < nTheta; ++i) {
99 for (int j = 0; j < nPhi; ++j) {
100 int j1 = (j + 1) % nPhi;
101
102 vtkNew<vtkTriangle> t1;
103 t1->GetPointIds()->SetId(0, grid[i][j]);
104 t1->GetPointIds()->SetId(1, grid[i + 1][j]);
105 t1->GetPointIds()->SetId(2, grid[i + 1][j1]);
106 triangles->InsertNextCell(t1);
107
108 vtkNew<vtkTriangle> t2;
109 t2->GetPointIds()->SetId(0, grid[i][j]);
110 t2->GetPointIds()->SetId(1, grid[i + 1][j1]);
111 t2->GetPointIds()->SetId(2, grid[i][j1]);
112 triangles->InsertNextCell(t2);
113 }
114 }
115
116 vtkNew<vtkPolyData> polyData;
117 polyData->SetPoints(points);
118 polyData->SetPolys(triangles);
119 polyData->GetPointData()->SetScalars(colors);
120
121 vtkNew<vtkPolyDataMapper> mapper;
122 mapper->SetInputData(polyData);
123
124 vtkNew<vtkActor> actor;
125 actor->SetMapper(mapper);
126 actor->GetProperty()->SetOpacity(opacity);
127 actor->GetProperty()->SetInterpolationToPhong();
128
129 return actor;
130}
