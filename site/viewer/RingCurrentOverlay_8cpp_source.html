36// Color by parent residue type (the ring knows its residue)
37static void ringColor(AminoAcid type, double& r, double& g, double& b) {
38 switch (type) {
39 case AminoAcid::PHE: r = 0.3; g = 0.9; b = 0.3; break;
40 case AminoAcid::TYR: r = 0.2; g = 0.85; b = 0.85; break;
41 case AminoAcid::TRP: r = 0.85; g = 0.3; b = 0.85; break;
42 default: r = 0.95; g = 0.9; b = 0.25; break; // HIS variants
43 }
44}

46void RingCurrentOverlay::buildActors(
47 const Protein& protein,
48 const ProteinConformation& conf)
49{
50 for (size_t i = 0; i < protein.RingCount(); ++i) {
51 const Ring& ring = protein.RingAt(i);
52 const RingGeometry& geo = conf.ring_geometries[i];
53
54 if (geo.radius < 0.01) continue;
55
56 AminoAcid resType = protein.ResidueAt(ring.parent_residue_index).type;
57
58 // Generate polygon vertices from center + normal + radius
59 constexpr int nVerts = 12;
60 Vec3 n = geo.normal.normalized();
61
62 Vec3 arbitrary = (std::abs(n.x()) < 0.9) ? Vec3(1, 0, 0) : Vec3(0, 1, 0);
63 Vec3 u = n.cross(arbitrary).normalized();
64 Vec3 v = n.cross(u);
65
66 // Translucent filled polygon — two copies offset above/below ring plane
67 for (int side = 0; side < 2; ++side) {
68 double sign = (side == 0) ? 1.0 : -1.0;
69 Vec3 offset = n * (0.15 * sign);
70
71 vtkNew<vtkPoints> pts;
72 vtkNew<vtkCellArray> polys;
73 vtkNew<vtkPolygon> polygon;
74 polygon->GetPointIds()->SetNumberOfIds(nVerts);
75 for (int vi = 0; vi < nVerts; ++vi) {
76 double angle = 2.0 * M_PI * vi / nVerts;
77 Vec3 p = geo.center + geo.radius * (u * std::cos(angle) + v * std::sin(angle)) + offset;
78 pts->InsertNextPoint(p.data());
79 polygon->GetPointIds()->SetId(vi, vi);
80 }
81 polys->InsertNextCell(polygon);
82
83 vtkNew<vtkPolyData> polyData;
84 polyData->SetPoints(pts);
85 polyData->SetPolys(polys);
86
87 vtkNew<vtkPolyDataMapper> mapper;
88 mapper->SetInputData(polyData);
89
90 vtkNew<vtkActor> actor;
91 actor->SetMapper(mapper);
92
93 double r, g, b;
94 ringColor(resType, r, g, b);
95 actor->GetProperty()->SetColor(r, g, b);
96 actor->GetProperty()->SetOpacity(0.35);
97 actor->GetProperty()->LightingOff();
98 actor->GetProperty()->SetEdgeVisibility(1);
99 actor->GetProperty()->SetEdgeColor(r * 0.7, g * 0.7, b * 0.7);
100 actor->GetProperty()->SetLineWidth(2.0);
101
102 renderer_->AddActor(actor);
103 actors_.push_back(actor);
104 }
105
106 // Small arrow along ring normal indicating current direction
107 vtkNew<vtkArrowSource> arrow;
108 arrow->SetTipResolution(16);
109 arrow->SetShaftResolution(16);
110 arrow->SetTipLength(0.3);
111 arrow->SetTipRadius(0.12);
112 arrow->SetShaftRadius(0.04);
113
114 vtkNew<vtkTransform> xform;
115 xform->PostMultiply();
116
117 Vec3 up = geo.normal;
118 Vec3 xAxis(1, 0, 0);
119 Vec3 rotAxis = xAxis.cross(up);
120 double rotAngle = std::acos(std::clamp(xAxis.dot(up), -1.0, 1.0))
121 * 180.0 / M_PI;
122
123 xform->Scale(1.2, 1.2, 1.2);
124 if (rotAxis.norm() > 1e-6) {
125 rotAxis.normalize();
126 xform->RotateWXYZ(rotAngle, rotAxis.x(), rotAxis.y(), rotAxis.z());
127 } else if (xAxis.dot(up) < 0) {
128 xform->RotateZ(180.0);
129 }
130 Vec3 arrowBase = geo.center + geo.normal * 0.3;
131 xform->Translate(arrowBase.x(), arrowBase.y(), arrowBase.z());
132
133 vtkNew<vtkTransformPolyDataFilter> xformFilter;
134 xformFilter->SetInputConnection(arrow->GetOutputPort());
135 xformFilter->SetTransform(xform);
136
137 vtkNew<vtkPolyDataMapper> arrowMapper;
138 arrowMapper->SetInputConnection(xformFilter->GetOutputPort());
139
140 vtkNew<vtkActor> arrowActor;
141 arrowActor->SetMapper(arrowMapper);
142
143 double r, g, b;
144 ringColor(resType, r, g, b);
145 arrowActor->GetProperty()->SetColor(r, g, b);
146 arrowActor->GetProperty()->SetOpacity(0.9);
147
148 renderer_->AddActor(arrowActor);
149 actors_.push_back(arrowActor);
150 }
151}
