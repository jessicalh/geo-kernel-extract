624
625 // Build vtkMolecule FIRST — RemoveAllViewProps clears the renderer,
626 // so this must happen before any overlay actors are added.
627 {
628 molecule_ = vtkSmartPointer<vtkMolecule>::New();
629
630 for (size_t i = 0; i < protein.AtomCount(); ++i) {
631 unsigned short anum = static_cast<unsigned short>(
632 AtomicNumberForElement(protein.AtomAt(i).element));
633 const Vec3& pos = conf.AtomAt(i).Position();
634 molecule_->AppendAtom(anum, pos.x(), pos.y(), pos.z());
635 }
636
637 for (size_t i = 0; i < protein.BondCount(); ++i) {
638 const auto& bond = protein.BondAt(i);
639 unsigned short vtkOrder = 1;
640 switch (bond.order) {
641 case BondOrder::Double: vtkOrder = 2; break;
642 case BondOrder::Triple: vtkOrder = 3; break;
643 case BondOrder::Aromatic: vtkOrder = 2; break; // show as double
644 case BondOrder::Peptide: vtkOrder = 1; break; // partial double, show single
645 default: break;
646 }
647 molecule_->AppendBond(bond.atom_index_a, bond.atom_index_b, vtkOrder);
648 }
649
650 molMapper_ = vtkSmartPointer<vtkOpenGLMoleculeMapper>::New();
651 molMapper_->SetInputData(molecule_);
652 molMapper_->UseBallAndStickSettings();
653
654 renderer_->RemoveAllViewProps();
655 molActor_ = vtkSmartPointer<vtkActor>::New();
656 molActor_->SetMapper(molMapper_);
657 renderer_->AddActor(molActor_);
658
659 udp_log("[diag] vtkMolecule built: %d atoms, %d bonds\n",
660 (int)molecule_->GetNumberOfAtoms(), (int)molecule_->GetNumberOfBonds());
661 }
662
663 // Overlays — added AFTER the molecule actor so they render on top.
664 if (ribbonOverlay_) { delete ribbonOverlay_; ribbonOverlay_ = nullptr; }
665 ribbonOverlay_ = new BackboneRibbonOverlay(renderer_, protein, conf);
666 ribbonOverlay_->setVisible(showRibbonCheck_->isChecked());
667
668 if (ringOverlay_) { delete ringOverlay_; ringOverlay_ = nullptr; }
669 ringOverlay_ = new RingCurrentOverlay(renderer_, protein, conf);
670
671 if (peptideBondOverlay_) { delete peptideBondOverlay_; peptideBondOverlay_ = nullptr; }
672 peptideBondOverlay_ = new PeptideBondOverlay(renderer_, protein, conf);
673 peptideBondOverlay_->setVisible(showPeptideBondsCheck_->isChecked());
674
675 // Butterfly overlay (B-field streamlines)
676 udp_log("[diag] butterfly fields: %zu grids\n", butterflyFields_.size());
677 if (!butterflyFields_.empty()) {
678 butterflyOverlay_ = new ButterflyOverlay(renderer_);
679 butterflyOverlay_->setData(butterflyFields_);
680 butterflyOverlay_->setVisible(showButterflyCheck_->isChecked());
681 }
682
683 // Field grid overlay — T0 shielding isosurface around each ring
684 if (!fieldGrids_.empty()) {
685 udp_log("[diag] %zu field grids, T0 range shown in status\n", fieldGrids_.size());
686 if (fieldGridOverlay_) { delete fieldGridOverlay_; fieldGridOverlay_ = nullptr; }
687 fieldGridOverlay_ = new FieldGridOverlay(renderer_);
688 double threshold = isoThresholdSlider_->value() / 100.0;
689 fieldGridOverlay_->setData(fieldGrids_, threshold, 0.35, 0);
690 fieldGridOverlay_->setVisible(showFieldGridCheck_->isChecked());
691 }
692
693 // Bond order color overlay — tubes colored by MOPAC Wiberg order
694 if (bondOrderActor_) {
695 renderer_->RemoveActor(bondOrderActor_);
696 bondOrderActor_ = nullptr;
697 }
698 if (conf.HasResult<MopacResult>()) {
699 const auto& mopac = conf.Result<MopacResult>();
700 const auto& topoOrders = mopac.TopologyBondOrders();
701
702 vtkNew<vtkPoints> pts;
703 vtkNew<vtkCellArray> lines;
704 vtkNew<vtkDoubleArray> scalars;
705 scalars->SetName("WibergOrder");
706 scalars->SetNumberOfComponents(1);
707
708 for (size_t i = 0; i < protein.BondCount(); ++i) {
709 double bo = (i < topoOrders.size()) ? topoOrders[i] : 0.0;
710 if (bo < 0.01) continue; // skip electronically insignificant bonds
711
712 const Bond& bond = protein.BondAt(i);
713 Vec3 posA = conf.AtomAt(bond.atom_index_a).Position();
714 Vec3 posB = conf.AtomAt(bond.atom_index_b).Position();
715 vtkIdType id0 = pts->InsertNextPoint(posA.data());
716 vtkIdType id1 = pts->InsertNextPoint(posB.data());
717 vtkNew<vtkLine> ln;
718 ln->GetPointIds()->SetId(0, id0);
719 ln->GetPointIds()->SetId(1, id1);
720 lines->InsertNextCell(ln);
721 scalars->InsertNextValue(bo);
722 }
723
724 if (lines->GetNumberOfCells() > 0) {
725 vtkNew<vtkPolyData> pd;
726 pd->SetPoints(pts);
727 pd->SetLines(lines);
728 pd->GetCellData()->SetScalars(scalars);
729
730 vtkNew<vtkTubeFilter> tube;
731 tube->SetInputData(pd);
732 tube->SetRadius(0.08);
733 tube->SetNumberOfSides(8);
734
735 // Cool-warm colormap: 0.8=blue(single) → 1.5=white(aromatic) → 2.0=red(double)
736 vtkNew<vtkLookupTable> lut;
737 lut->SetNumberOfTableValues(256);
738 lut->SetRange(0.8, 2.1);
739 lut->SetHueRange(0.65, 0.0); // blue → red
740 lut->SetSaturationRange(0.7, 0.9);
741 lut->SetValueRange(0.8, 0.95);
742 lut->Build();
743
744 vtkNew<vtkPolyDataMapper> mapper;
745 mapper->SetInputConnection(tube->GetOutputPort());
746 mapper->SetScalarModeToUseCellData();
747 mapper->SetLookupTable(lut);
748 mapper->SetScalarRange(0.8, 2.1);
749
750 bondOrderActor_ = vtkSmartPointer<vtkActor>::New();
751 bondOrderActor_->SetMapper(mapper);
752 bondOrderActor_->GetProperty()->SetOpacity(0.7);
753 bondOrderActor_->SetVisibility(0); // off by default, toggled by checkbox
754 renderer_->AddActor(bondOrderActor_);
755
756 udp_log("[diag] Bond order overlay: %lld bonds with BO >= 0.01\n",
757 (long long)lines->GetNumberOfCells());
758 }
759 }
760
761 renderer_->ResetCamera();
762 renderWindow_->Render();
763
764 // Status: read tier classification from library (set by PredictionResult)
765 int nReport = 0, nPass = 0, nSilent = 0;
766 for (size_t i = 0; i < conf.AtomCount(); ++i) {
767 switch (conf.AtomAt(i).tier) {
768 case HeuristicTier::REPORT: nReport++; break;
769 case HeuristicTier::PASS: nPass++; break;
770 case HeuristicTier::SILENT: nSilent++; break;
771 }
772 }
773
774 udp_log("[diag] setting status text\n");
775 QString status = QString("Loaded %1: %2 atoms (%3 REPORT, %4 PASS, %5 SILENT), %6 rings")
776 .arg(QString::fromStdString(currentProteinId_))
777 .arg(protein.AtomCount())
778 .arg(nReport)
779 .arg(nPass)
780 .arg(nSilent)
781 .arg(protein.RingCount());
782 statusLabel_->setText(status);
783
784 udp_log("[diag] calling updateOverlay\n");
785 updateOverlay();
786 udp_log("[diag] onComputeFinished done\n");
787}

895void MainWindow::pickAtom(int displayX, int displayY) {
896 if (!protein_) return;
897
898 // Original session-4 ray casting, with DPR correction and logging.
899 double dpr = vtkWidget_->devicePixelRatioF();
900 int vtkX = static_cast<int>(displayX * dpr);
901 int vtkY = static_cast<int>((vtkWidget_->height() - displayY) * dpr);
902
903 udp_log("[pick] pickAtom qt=(%d,%d) vtk=(%d,%d) dpr=%.1f winSz=(%d,%d)\n",
904 displayX, displayY, vtkX, vtkY, dpr,
905 renderWindow_->GetSize()[0], renderWindow_->GetSize()[1]);
906
907 auto* camera = renderer_->GetActiveCamera();
908 double camPos[3];
909 camera->GetPosition(camPos);
910 Vec3 rayOrigin(camPos[0], camPos[1], camPos[2]);
911
912 renderer_->SetDisplayPoint(vtkX, vtkY, 0.0);
913 renderer_->DisplayToWorld();
914 double worldPt[4];
915 renderer_->GetWorldPoint(worldPt);
916 Vec3 clickWorld(worldPt[0] / worldPt[3],
917 worldPt[1] / worldPt[3],
918 worldPt[2] / worldPt[3]);
919
920 Vec3 rayDir = (clickWorld - rayOrigin).normalized();
921
922 udp_log("[pick] ray origin=(%.2f,%.2f,%.2f) dir=(%.3f,%.3f,%.3f)\n",
923 rayOrigin.x(), rayOrigin.y(), rayOrigin.z(),
924 rayDir.x(), rayDir.y(), rayDir.z());
925
926 const auto& conf = protein_->Conformation();
927 double bestDist = 1e30;
928 int bestAtom = -1;
929 for (size_t i = 0; i < conf.AtomCount(); ++i) {
930 Vec3 pos = conf.AtomAt(i).Position();
931 Vec3 toAtom = pos - rayOrigin;
932 double projLen = toAtom.dot(rayDir);
933 if (projLen < 0) continue;
934 Vec3 closest = rayOrigin + projLen * rayDir;
935 double dist = (pos - closest).norm();
936 if (dist < bestDist) {
937 bestDist = dist;
938 bestAtom = static_cast<int>(i);
939 }
940 }
941
942 udp_log("[pick] best atom=%d ray-dist=%.3f A\n", bestAtom, bestDist);
943
944 if (bestAtom >= 0 && bestDist < 2.0) {
945 size_t atomIndex = static_cast<size_t>(bestAtom);
946 const auto& id = protein_->AtomAt(atomIndex);
947 const auto& res = protein_->ResidueAt(id.residue_index);
948 Vec3 pos = conf.AtomAt(atomIndex).Position();
949
950 udp_log("[pick] → atom %zu: %s %s-%d\n", atomIndex,
951 id.pdb_atom_name.c_str(),
952 ThreeLetterCodeForAminoAcid(res.type).c_str(),
953 res.sequence_number);
954
955 populateAtomInfo(atomIndex);
956 populateAtomBonds(atomIndex);
957 populateGeometryChoices(atomIndex);
958 atomInfoDock_->raise();
959
960 if (selectionActor_) renderer_->RemoveActor(selectionActor_);
961 vtkNew<vtkSphereSource> sphere;
962 sphere->SetCenter(pos.x(), pos.y(), pos.z());
963 sphere->SetRadius(1.0);
964 sphere->SetPhiResolution(16);
965 sphere->SetThetaResolution(16);
966 vtkNew<vtkPolyDataMapper> mapper;
967 mapper->SetInputConnection(sphere->GetOutputPort());
968 selectionActor_ = vtkSmartPointer<vtkActor>::New();
969 selectionActor_->SetMapper(mapper);
970 selectionActor_->GetProperty()->SetColor(1.0, 1.0, 0.0);
971 selectionActor_->GetProperty()->SetOpacity(0.3);
972 renderer_->AddActor(selectionActor_);
973 renderWindow_->Render();
974
975 statusLabel_->setText(QString("Atom %1: %2 %3-%4")
976 .arg(atomIndex)
977 .arg(QString::fromStdString(id.pdb_atom_name))
978 .arg(QString::fromStdString(ThreeLetterCodeForAminoAcid(res.type)))
979 .arg(res.sequence_number));
980 }
981}
