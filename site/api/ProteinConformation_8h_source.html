66 template<typename T>
67 T& Result() {
68 auto it = results_.find(std::type_index(typeid(T)));
69 if (it == results_.end()) {
70 // Diagnostic: state what is attached, what is missing
71 std::ostringstream msg;
72 msg << "Result<" << typeid(T).name() << "> not attached. Attached: [";
73 bool first = true;
74 for (const auto& kv : results_) {
75 if (!first) msg << ", ";
76 msg << kv.second->Name();
77 first = false;
78 }
79 msg << "]";
80 // Return codes, not exceptions -- but Result<T> is a contract guarantee.
81 // If called when not attached, this is a programming error.
82 // For now, we abort with a diagnostic message.
83 fprintf(stderr, "FATAL: %s\n", msg.str().c_str());
84 std::abort();
85 }
86 return static_cast<T&>(*it->second);
87 }
