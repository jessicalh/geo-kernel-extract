2//
3// MutationDeltaResult: ConformationResult that stores the per-atom deltas
4// between this (WT) conformation and a mutant conformation.
5//
6// Attaches to the WT conformation. The mutant conformation is an extra
7// input to Compute, not a dependency. Stores atom correspondence and
8// per-atom delta tensors (DFT shielding, APBS fields, MOPAC charges,
9// DSSP, graph, charges) internally — does NOT write to ConformationAtom.
10//
11// Ring proximity: for each matched atom, stores cylindrical coordinates
12// relative to each removed ring (the mutation site rings). Uses SVD-based
13// ring normals from GeometryResult, cylindrical decomposition (z, rho,
14// theta), McConnell factor, and exponential decay weighting.
15//

37// Cylindrical coordinates of an atom relative to one removed ring
38struct RingProximity {
39 size_t ring_index = 0;
40 RingTypeIndex ring_type = RingTypeIndex::PheBenzene;
41 double distance = 0.0; // |d| to ring center (Angstroms)
42 double z = 0.0; // height above ring plane (signed)
43 double rho = 0.0; // radial distance in ring plane
44 double theta = 0.0; // cone angle from normal: atan2(rho, z)
45 double mcconnell_factor = 0.0; // (3cos^2 theta - 1) / r^3
46 double exp_decay = 0.0; // exp(-distance / 4.0)
47};
