2//
3// GeometryChoice: runtime record of one geometric decision made by a calculator.
4//
5// A GeometryChoice is a bag of model objects — ConformationAtom*, Ring*, Bond*,
6// and named numbers — recording what a calculator decided, which entities were
7// involved, and whether each was included or excluded.
8//
9// The conformation owns a flat vector<GeometryChoice>. Calculators populate it
10// during Compute() via the GeometryChoiceBuilder factory, which enforces that
11// every choice is populated within a lambda (keeping the recording code visually
12// separate from the physics).
13//
14// The UI walks the list, follows pointers back into the live model, and draws.
15//
16// Entity roles:
17// Source — the ring/bond/atom generating the field
18// Target — the atom being shielded or evaluated
19// Context — referenced but not directly source or target (e.g. bond endpoints)
20//
21// Outcome:
22// Included — entity passed the gate
23// Excluded — entity failed the gate
24// Triggered — a guard or clamp fired (singularity, sanity, refinement)
25// NotTriggered — a guard was checked but did not fire
26//
27

130// ============================================================================
131// GeometryChoiceBuilder — factory that enforces population in a lambda.
132//
133// Usage in a calculator's Compute():
134//
135// GeometryChoiceBuilder choices(conf);
136//
137// choices.Record(CalculatorId::BiotSavart, group, "ring horizon",
138// [&](GeometryChoice& gc) {
139// AddRing(gc, ring, EntityRole::Source, EntityOutcome::Included);
140// AddAtom(gc, &conf.AtomAt(ai), ai, EntityRole::Target, EntityOutcome::Included);
141// AddNumber(gc, "horizon", 15.0, "A");
142// AddNumber(gc, "distance", dist, "A");
143// });
144//
145// The lambda body is the ONLY place entities and numbers are added.
146// This keeps recording code visually offset from physics code.
147// ============================================================================
148
149class GeometryChoiceBuilder {
150public:
151 explicit GeometryChoiceBuilder(ProteinConformation& conf);
152
153 // Record one geometric decision. The populate lambda fills the choice,
154 // then it goes straight onto the conformation's list. No commit step.
155 void Record(CalculatorId calculator,
156 size_t group_key,
157 const char* label,
158 std::function<void(GeometryChoice&)> populate);
159
160private:
161 ProteinConformation& conf_;
162};
