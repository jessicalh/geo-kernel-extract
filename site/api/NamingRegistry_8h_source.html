2//
3// NamingRegistry: the single authority for residue and atom name translation.
4//
5// Every tool in the pipeline has its own naming universe:
6// PDB/Standard: HIS, CYS, H (backbone amide)
7// AMBER: HID/HIE/HIP, CYX/CYM, H
8// CHARMM36m: HSD/HSE/HSP, CYS2/CYM, HN
9// ORCA: standard PDB names (numbered atoms)
10// OpenMM: AMBER-like (HIE, HID, CYX)
11//
12// This registry translates between them. It is the GATEKEEPER:
13// unknown names are rejected. No guessing. No silent pass-through.
14//
15// After translation, typed objects carry their properties. The registry
16// exists so that the rest of the system NEVER needs strings for identity.
17//

27// The tools we talk to. Each has its own naming conventions.
28enum class ToolContext {
29 Standard, // PDB / IUPAC canonical names
30 Amber, // AMBER force fields (ff14SB, ff19SB): HID/HIE/HIP, CYX, ASH, GLH
31 Charmm, // CHARMM36m: HSD/HSE/HSP, CYS2, ASPP, GLUP, HN for backbone amide
32 Orca, // ORCA DFT (standard PDB names, numbered atoms)
33 OpenMM, // OpenMM (AMBER-like: HIE, HID, CYX)
34};
35
36inline const char* ToolContextName(ToolContext ctx) {
37 switch (ctx) {
38 case ToolContext::Standard: return "Standard";
39 case ToolContext::Amber: return "Amber";
40 case ToolContext::Charmm: return "CHARMM";
41 case ToolContext::Orca: return "ORCA";
42 case ToolContext::OpenMM: return "OpenMM";
43 }
44 return "Unknown";
45}
