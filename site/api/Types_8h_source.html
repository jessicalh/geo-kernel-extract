217// ============================================================================
218// SphericalTensor -- irreducible decomposition of a 3x3 tensor
219//
220// Any 3x3 tensor sigma decomposes uniquely into three parts:
221// T0: scalar (trace/3) -- the isotropic component
222// T1: pseudovector (antisymmetric part) -- 3 components
223// T2: traceless symmetric tensor -- 5 components (m = -2..+2)
224//
225// T2 uses isometric normalization (real spherical harmonics).
226// This preserves the L2 norm: sum(|T2_m|^2) == sum(S_ij^2).
227// ============================================================================
228
229struct SphericalTensor {
230 double T0 = 0.0;
231 std::array<double, 3> T1 = {};
232 std::array<double, 5> T2 = {};
233
234 double Isotropic() const { return T0; }
235 const std::array<double, 3>& Antisymmetric() const { return T1; }
236 const std::array<double, 5>& TracelessSymmetric() const { return T2; }
237
238 // Decompose a 3x3 tensor into T0 + T1 + T2.
239 static SphericalTensor Decompose(const Mat3& tensor);
240
241 // Reconstruct the 3x3 tensor from T0, T1, T2.
242 Mat3 Reconstruct() const;
243
244 // L2 norm of the T2 components.
245 double T2Magnitude() const;
246};
