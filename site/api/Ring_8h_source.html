2//
3// Ring type class hierarchy.

5// Ring types ARE classes with physics properties baked in.
6// Each type provides const properties derived from its identity.
7// Calculator code is ring-type-agnostic: ring.Intensity(), ring.JBLobeOffset().

10// SixMemberedRing: PheBenzeneRing, TyrPhenolRing, TrpBenzeneRing
11// FiveMemberedRing: TrpPyrroleRing, HisImidazoleRing, HidImidazoleRing, HieImidazoleRing
12// FusedRing: IndolePerimeterRing (TRP 9-atom)

45// ============================================================================
46// Ring (base class)
47// ============================================================================
48
49class Ring {
50public:
51 // Structural identity (topology, set at construction)
52 std::vector<size_t> atom_indices;
53 RingTypeIndex type_index = RingTypeIndex::PheBenzene;
54 size_t parent_residue_index = 0;
55 int parent_residue_number = 0;
56 size_t fused_partner_index = SIZE_MAX;
57
58 virtual ~Ring() = default;
59
60 // Virtual const properties (overridden by each type class)
61 virtual double Intensity() const = 0;
62 virtual double LiteratureIntensity() const = 0;
63 virtual double JBLobeOffset() const = 0;
64 virtual int NitrogenCount() const = 0;
65 virtual RingAromaticity Aromaticity() const = 0;
66 virtual int RingSizeValue() const = 0;
67 virtual const char* TypeName() const = 0;
68
69 // Non-virtual queries
70 bool IsFused() const { return fused_partner_index != SIZE_MAX; }
71 int TypeIndexAsInt() const { return static_cast<int>(type_index); }
72
73 // Compute geometry from positions (SVD normal)
74 RingGeometry ComputeGeometry(const std::vector<Vec3>& positions) const;
75
76 // Accumulated properties (set during extraction passes)
77 RingAccumulated accumulated;
78};
