2//
3// ChargeSource: typed abstraction for where per-atom charges come from.
4//
5// The force field determines charges, naming, and VdW parameters.
6// Different force fields produce different numbers for the same atom.
7// This is NOT a string dispatch — each source is a distinct type that
8// knows how to provide charges for a Protein's atoms.
9//
10// Implementations:
11// ParamFileChargeSource — ff14SB from flat parameter file (current)
12// GmxTprChargeSource — CHARMM36m (or any FF) from GROMACS .tpr
13// PrmtopChargeSource — AMBER prmtop (future, needs cpptraj)
14// StubChargeSource — uniform test charges
15//
16// ChargeAssignmentResult::Compute takes a ChargeSource, not a path string.
17//

29// Force field identity. Recorded in ProteinBuildContext as provenance.
30enum class ForceField {
31 Amber_ff14SB,
32 Amber_ff19SB,
33 CHARMM36m,
34 Unknown
35};
36
37inline const char* ForceFieldName(ForceField ff) {
38 switch (ff) {
39 case ForceField::Amber_ff14SB: return "ff14SB";
40 case ForceField::Amber_ff19SB: return "ff19SB";
41 case ForceField::CHARMM36m: return "CHARMM36m";
42 case ForceField::Unknown: return "unknown";
43 }
44 return "unknown";
45}

48// Per-atom charge and radius from whatever source.
49struct AtomChargeRadius {
50 double charge = 0.0; // elementary charges (e)
51 double radius = 0.0; // Angstroms (VdW radius for APBS)
52};

55// Abstract charge source. Each implementation knows how to read
56// charges from its data format and match them to protein atoms.
57class ChargeSource {
58public:
59 virtual ~ChargeSource() = default;
60
61 // Which force field produced these charges.
62 virtual ForceField SourceForceField() const = 0;
63
64 // Human-readable description for logging.
65 virtual std::string Describe() const = 0;
66
67 // Load charges for all atoms in this protein/conformation.
68 // Returns one entry per atom (parallel to protein's atom list).
69 // Empty vector on failure (error reported via error_out).
70 virtual std::vector<AtomChargeRadius> LoadCharges(
71 const Protein& protein,
72 const ProteinConformation& conf,
73 std::string& error_out) const = 0;
74};

153private:
154 std::string path_;
155 ForceField ff_;
