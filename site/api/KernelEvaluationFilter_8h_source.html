2//
3// KernelEvaluationFilter: ABC and concrete filters for deciding whether
4// a geometric kernel should be evaluated at a given atom-source pair.
5//
6// Every calculator that evaluates a geometric kernel (dipolar, Coulomb,
7// surface integral) holds a KernelFilterSet. Before computing the kernel
8// at a field point, the calculator builds a KernelEvaluationContext from
9// the evaluation geometry and passes it to the filter set. All filters
10// must accept for the kernel to be evaluated.
11//
12// Filters operate on first-principles physics criteria:
13// - DipolarNearFieldFilter: multipole expansion invalid inside source
14// - SelfSourceFilter: field undefined at the source itself
15// - SequentialExclusionFilter: through-bond coupling not modelled
16//
17// Type-based filters (SelfSourceFilter) are permitted when the physics
18// reason is explicit: an atom that IS the source cannot be a field point
19// for that source's dipolar field.
20//
21// Filter sets are configurable from TOML. The test for a filter is:
22// it works but not as well without it. A filter that changes nothing
23// is dead code. A filter without which the calculator crashes is
24// papering over a bug.
25//

40// ============================================================================
41// KernelEvaluationContext: the geometric facts about one potential
42// kernel evaluation. Assembled by the calculator from already-computed
43// distances and topology. No new spatial computation happens here.
44// ============================================================================
45
46struct KernelEvaluationContext {
47 // Distance from field point to source expansion center (Angstroms).
48 // Already computed as part of the kernel setup.
49 double distance = 0.0;
50
51 // Spatial extent of the source distribution (Angstroms).
52 // Bond: bond length. H-bond: N...O distance. Ring: ring diameter.
53 // Zero if the source is a point (single atom in Coulomb sum).
54 double source_extent = 0.0;
55
56 // Sequence separation between field-point residue and source residue.
57 // -1 if not applicable or not known.
58 int sequence_separation = -1;
59
60 // Field-point atom index.
61 size_t atom_index = SIZE_MAX;
62
63 // Source atom indices. For a bond: the two endpoints. For an H-bond:
64 // donor N and acceptor O. For a ring: SIZE_MAX (no single atom).
65 // For a Coulomb pair: source_atom_a = the source atom, source_atom_b = SIZE_MAX.
66 size_t source_atom_a = SIZE_MAX;
67 size_t source_atom_b = SIZE_MAX;
68
69 // Source ring index in the Protein's ring list. SIZE_MAX if the
70 // source is not a ring (bonds, H-bonds, Coulomb pairs).
71 // Used by RingBondedExclusionFilter to check whether the field-point
72 // atom is a vertex of this ring or bonded to one.
73 size_t source_ring_index = SIZE_MAX;
74
75 // Optional: the conformation that owns the evaluation geometry.
76 // Available for future physics-aware filters that need to query
77 // positions, bond directions, or already-computed results at
78 // neighbouring atoms. Do NOT use this for simple distance/index
79 // checks — use the fields above. This exists so that the mechanism
80 // for richer filters stays within the object model rather than
81 // requiring a parallel data path.
82 //
83 // This is safe to use: calculators run in dependency order and
84 // all declared dependencies are attached before Compute() runs.
85 // Do NOT add order-checking machinery to this library. The
86 // dependency graph already guarantees safety.
87 const ProteinConformation* conf = nullptr;
88};

118// ============================================================================
119// DipolarNearFieldFilter
120//
121// Physics: the dipolar kernel K_ab = (3 d̂_a d̂_b - δ_ab) / r³ models
122// the source as a point at the expansion center. When the field point
123// is closer to the center than the source radius (half the source
124// extent), the multipole expansion does not converge. The kernel value
125// is large but physically meaningless — the field point is inside the
126// charge/current distribution.
127//
128// Criterion: distance > source_extent / 2.
129// Source extent is a measured property of each source, not a global
130// tuning parameter.
131//
132// Demonstrated effect (HBondResult, 466 proteins):
133// With filter: max |T2| = 0.78 A^-3
134// Without: max |T2| = 1908 A^-3 (atoms 0.2A from H-bond midpoint)
135// ============================================================================
136
137// ============================================================================
138// MinDistanceFilter
139//
140// Physics: at distances below 0.1 A the field point is essentially on top
141// of the source. The 1/r^n kernels diverge numerically. The kernel helpers
142// (ComputeBondKernel, AccumulateTensor, etc.) also guard internally, but
143// this filter prevents the evaluation from starting at all and provides
144// a named rejection for GeometryChoice recording.
145//
146// Criterion: distance >= MIN_DISTANCE (0.1 A).
147// ============================================================================
148
149class MinDistanceFilter : public KernelEvaluationFilter {
150public:
151 bool Accept(const KernelEvaluationContext& ctx) const override {
152 return ctx.distance >= MIN_DISTANCE;
153 }
154
155 const char* Name() const override {
156 return "MinDistanceFilter";
157 }
158
159 const char* Description() const override {
160 return "Singularity guard: 1/r^n kernels diverge below 0.1 A";
161 }
162
163 std::string RejectReason(const KernelEvaluationContext& ctx) const override {
164 return std::string("MinDistanceFilter: atom=")
165 + std::to_string(ctx.atom_index)
166 + " distance=" + std::to_string(ctx.distance)
167 + "A < " + std::to_string(MIN_DISTANCE) + "A";
168 }
169};

198// ============================================================================
199// SelfSourceFilter
200//
201// Physics: the dipolar field of a source is undefined at the source
202// itself. An atom that IS an endpoint of a bond cannot be a field point
203// for that bond's McConnell kernel. An atom that IS the donor N or
204// acceptor O cannot be a field point for that H-bond's kernel.
205//
206// This is type-based: it checks atom indices, not distances. The physics
207// justification is that the source atom's own electron cloud generates
208// the field — it does not experience its own dipolar field as an
209// external perturbation.
210//
211// Criterion: atom_index != source_atom_a AND atom_index != source_atom_b.
212// ============================================================================
213
214class SelfSourceFilter : public KernelEvaluationFilter {
215public:
216 bool Accept(const KernelEvaluationContext& ctx) const override {
217 if (ctx.atom_index == ctx.source_atom_a) return false;
218 if (ctx.source_atom_b != SIZE_MAX && ctx.atom_index == ctx.source_atom_b)
219 return false;
220 return true;
221 }
222
223 const char* Name() const override {
224 return "SelfSourceFilter";
225 }
226
227 const char* Description() const override {
228 return "Dipolar field undefined at the source atom itself "
229 "(atom cannot be field point for its own source)";
230 }
231
232 std::string RejectReason(const KernelEvaluationContext& ctx) const override {
233 size_t match = (ctx.atom_index == ctx.source_atom_a)
234 ? ctx.source_atom_a : ctx.source_atom_b;
235 return std::string("SelfSourceFilter: atom=")
236 + std::to_string(ctx.atom_index)
237 + " IS source_atom=" + std::to_string(match);
238 }
239};

242// ============================================================================
243// SequentialExclusionFilter
244//
245// Physics: atoms within a few residues in sequence share through-bond
246// electronic coupling (inductive, hyperconjugation) that the through-
247// space dipolar model does not capture. The dipolar kernel gives a
248// geometric prediction at these distances, but the actual shielding
249// perturbation is dominated by through-bond effects that have a
250// different angular structure.
251//
252// Criterion: sequence_separation >= min_separation.
253// The default min_separation (2) excludes same-residue and immediate
254// neighbour interactions. Configurable from TOML.
255// ============================================================================
256
257class SequentialExclusionFilter : public KernelEvaluationFilter {
258public:
259 explicit SequentialExclusionFilter(int min_separation = 2)
260 : min_separation_(min_separation) {}
261
262 bool Accept(const KernelEvaluationContext& ctx) const override {
263 if (ctx.sequence_separation < 0) return true; // unknown, allow
264 return ctx.sequence_separation >= min_separation_;
265 }
266
267 const char* Name() const override {
268 return "SequentialExclusionFilter";
269 }
270
271 const char* Description() const override {
272 return "Through-bond coupling dominates through-space dipolar "
273 "at small sequence separations (dipolar model inapplicable)";
274 }
275
276 std::string RejectReason(const KernelEvaluationContext& ctx) const override {
277 return std::string("SequentialExclusionFilter: atom=")
278 + std::to_string(ctx.atom_index)
279 + " seq_sep=" + std::to_string(ctx.sequence_separation)
280 + " < min=" + std::to_string(min_separation_);
281 }
282
283 int MinSeparation() const { return min_separation_; }
284
285private:
286 int min_separation_;
287};

290// ============================================================================
291// RingBondedExclusionFilter
292//
293// Physics: an atom that is a vertex of a ring, or covalently bonded to a
294// ring vertex, is in the through-bond regime for that ring's field. The
295// through-space multipole expansion (dipolar, quadrupole, surface integral,
296// dispersion) does not model through-bond electronic coupling. Evaluating
297// the kernel at these atoms gives a large number that is physically
298// meaningless — the atom is part of the source distribution.
299//
300// This is a topological check, not a distance proxy. It uses the bond
301// graph to determine membership. Compare to DipolarNearFieldFilter which
302// catches most of the same atoms by distance, but has a boundary case
303// for ring atoms that sit exactly at the filter threshold.
304//
305// Construction: takes a const reference to the Protein whose topology
306// defines which atoms are ring vertices and what they are bonded to.
307// At construction, walks Protein → Ring → atom_indices → bond_indices
308// to build a per-ring exclusion set. This is the one-way path from
309// the formal protein definition down through typed topology.
310//
311// Criterion: atom_index NOT in excluded set for source_ring_index.
312// Requires source_ring_index to be set on the evaluation context.
313// ============================================================================
314
315class RingBondedExclusionFilter : public KernelEvaluationFilter {
316public:
317 // Walks the Protein's topology once to build per-ring exclusion sets.
318 // Each set contains the ring's vertices and all atoms bonded to them.
319 explicit RingBondedExclusionFilter(const Protein& protein);
320
321 bool Accept(const KernelEvaluationContext& ctx) const override {
322 if (ctx.source_ring_index == SIZE_MAX) return true; // not a ring source
323 if (ctx.source_ring_index >= ring_bonded_.size()) return true;
324 return ring_bonded_[ctx.source_ring_index].count(ctx.atom_index) == 0;
325 }
326
327 const char* Name() const override {
328 return "RingBondedExclusionFilter";
329 }
330
331 const char* Description() const override {
332 return "Through-space kernel invalid for ring vertices and their "
333 "bonded neighbours (through-bond regime, not through-space)";
334 }
335
336 std::string RejectReason(const KernelEvaluationContext& ctx) const override {
337 return std::string("RingBondedExclusionFilter: atom=")
338 + std::to_string(ctx.atom_index)
339 + " is bonded to ring " + std::to_string(ctx.source_ring_index);
340 }
341
342private:
343 std::vector<std::set<size_t>> ring_bonded_;
344};

347// ============================================================================
348// KernelFilterSet: a calculator's ordered list of filters.
349// All must accept for the kernel to be evaluated.
350// ============================================================================
351
352class KernelFilterSet {
353public:
354 void Add(std::unique_ptr<KernelEvaluationFilter> filter) {
355 filters_.push_back(std::move(filter));
356 rejection_counts_.push_back(0);
357 }
358
359 // Returns true if all filters accept. Tracks per-filter rejection counts
360 // and logs each rejection with the evaluation context.
361 bool AcceptAll(const KernelEvaluationContext& ctx) {
362 for (size_t i = 0; i < filters_.size(); ++i) {
363 if (!filters_[i]->Accept(ctx)) {
364 rejection_counts_[i]++;
365 last_rejector_ = filters_[i]->Name();
366 if (log_rejections_) {
367 rejections_.push_back({
368 filters_[i]->Name(),
369 ctx.atom_index,
370 ctx.source_atom_a,
371 ctx.source_atom_b,
372 ctx.distance,
373 ctx.source_extent
374 });
375 rejection_reasons_.push_back(
376 filters_[i]->RejectReason(ctx));
377 }
378 return false;
379 }
380 }
381 return true;
382 }
383
384 // Enable/disable per-rejection logging. Off by default for batch
385 // performance. Enable for single-protein diagnostics.
386 void SetLogRejections(bool enable) { log_rejections_ = enable; }
387
388 // Number of filters in the set.
389 size_t Size() const { return filters_.size(); }
390
391 // Name of the filter that rejected the most recent AcceptAll() call.
392 // Only meaningful after AcceptAll() returns false.
393 const char* LastRejectorName() const { return last_rejector_; }
394
395 // Access for logging/reporting.
396 const KernelEvaluationFilter& FilterAt(size_t i) const {
397 return *filters_[i];
398 }
399
400 // How many evaluations this filter rejected.
401 int RejectionCount(size_t i) const {
402 return rejection_counts_[i];
403 }
404
405 // Total rejections across all filters.
406 int TotalRejections() const {
407 int total = 0;
408 for (int c : rejection_counts_) total += c;
409 return total;
410 }
411
412 // Report all filter names (for startup logging).
413 std::string Describe() const {
414 std::string s;
415 for (size_t i = 0; i < filters_.size(); ++i) {
416 if (i > 0) s += ", ";
417 s += filters_[i]->Name();
418 }
419 return s;
420 }
421
422 // Report rejection counts per filter (for completion logging).
423 std::string ReportRejections() const {
424 std::string s;
425 for (size_t i = 0; i < filters_.size(); ++i) {
426 if (i > 0) s += ", ";
427 s += filters_[i]->Name();
428 s += "=" + std::to_string(rejection_counts_[i]);
429 }
430 return s;
431 }
