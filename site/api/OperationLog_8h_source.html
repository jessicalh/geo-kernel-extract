2//
3// OperationLog: structured UDP logging for every meaningful operation.
4//
5// This is not optional debugging. This is the system's external memory.
6// When an agent builds layers of complexity and can no longer hold the
7// full state in context, the operation log shows what actually happened.
8//
9// Every operation that changes state, computes a result, loads data,
10// calls an external tool, or encounters an error emits a structured
11// JSON message over UDP. A listener captures the stream.
12//
13// The log is ALWAYS ON. The cost of a UDP send is negligible compared
14// to any calculation this library performs. The cost of NOT logging
15// is hours of debugging silent failures.
16//
17// Message format (JSON over UDP or stderr):
18// {
19// "ts": "2026-04-01T18:34:56.789",
20// "level": "info",
21// "op": "DsspResult::Compute",
22// "detail": "protein=1UBQ, residues=76, ss_assigned=76"
23// }
24//

32// Log channel bitmask. Set in TOML [logging].channels.
33// Warnings and errors ALWAYS log regardless of channel mask.
34// Info-level logs only fire if their channel bit is set.
35enum LogChannel : uint32_t {
36 LogDiag0 = 1 << 0, // Reserved: diagnostic slot 0
37 LogDiag1 = 1 << 1, // Reserved: diagnostic slot 1
38 LogBondClassify = 1 << 2, // Bond classification (per-bond, chatty)
39 LogNaming = 1 << 3, // NamingRegistry translations
40 LogRingDetect = 1 << 4, // Ring detection and geometry
41 LogCalcDispatch = 1 << 5, // Calculator dispatch
42 LogCalcBiotSavart = 1 << 6, // BiotSavart detail
43 LogCalcHaighMal = 1 << 7, // HaighMallion detail
44 LogCalcMcConnell = 1 << 8, // McConnell detail
45 LogCalcOther = 1 << 9, // PQ, RSA, LD, HB, Coulomb
46 LogFileIO = 1 << 10, // PDB/XYZ/XTC file reading
47 LogDatabase = 1 << 11, // Database queries
48 LogAtomMapping = 1 << 12, // Atom correspondence / WT-ALA matching
49 LogDSSP = 1 << 13, // DSSP computation
50 LogProtonation = 1 << 14, // Protonation detection
51 LogConformation = 1 << 15, // Conformation add/modify
52 LogResultAttach = 1 << 16, // ConformationResult attachment
53 LogAPBS = 1 << 17, // APBS electrostatics
54 LogToolExec = 1 << 18, // External tool execution
55 LogMopac = 1 << 19, // MOPAC computation
56 LogCharges = 1 << 20, // Charge assignment
57 LogViewer = 1 << 21, // Viewer events
58 LogAll = 0xFFFFFFFF
59};
