27// A protonation variant (e.g., HID for histidine, ASH for aspartate).
28//
29// VARIANT INDEX CONTRACT: the position of each variant in
30// AminoAcidType::variants is load-bearing. ProtonationDetectionResult,
31// ChargeAssignmentResult, and ProtonationState all use variant_index
32// to identify the protonation state. Reordering silently breaks every
33// protonation assignment in the system.
34//
35// Canonical indices (asserted at startup in ValidateVariantIndices):
36// HIS: 0=HID (delta), 1=HIE (epsilon), 2=HIP (doubly)
37// ASP: 0=ASH (protonated)
38// GLU: 0=GLH (protonated)
39// CYS: 0=CYX (disulfide), 1=CYM (deprotonated)
40// LYS: 0=LYN (deprotonated)
41// ARG: 0=ARN (deprotonated, pKa ~12.5, very rare)
42// TYR: 0=TYM (deprotonated)
43//
44struct ProtonationVariant {
45 const char* name; // AMBER-standard name: "HID", "ASH", "CYX"
46 const char* description; // Human-readable: "Nd-protonated (delta)"
47 int formal_charge; // Formal charge in this state
48 const char* registry_key; // NamingRegistry semantic key: "delta", "protonated"
49 // Used by ResolveForTool(canonical, context, key)
50};
